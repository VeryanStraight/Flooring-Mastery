CREATE TABLE Taxes (
    State CHAR(2) PRIMARY KEY,
    StateName VARCHAR(100),
    TaxRate NUMERIC(5, 2)
);

CREATE TABLE Products (
    ProductType VARCHAR(50) PRIMARY KEY,
    CostPerSquareFoot NUMERIC(10, 2),
    LaborCostPerSquareFoot NUMERIC(10, 2)
);

CREATE TABLE Orders (
    date DATE,
    OrderNumber SERIAL,
    CustomerName VARCHAR(255),
    State CHAR(2),
    ProductType VARCHAR(50),
    Area NUMERIC(10, 2),
    MaterialCost NUMERIC(10, 4),
    LaborCost NUMERIC(10, 4),
    Tax NUMERIC(10, 6),
    Total NUMERIC(10, 6),
    PRIMARY KEY (date, OrderNumber),
    FOREIGN KEY (State) REFERENCES Taxes(State),
    FOREIGN KEY (ProductType) REFERENCES Products(ProductType)
);
